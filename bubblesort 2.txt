void bubbleSort2(int x[])
{
	int aux = 0;
	//ordenando de forma crescente
	//la�o com a quantidade de elementos do vetor - 1
	for (int j = 1; j <= 4; j++)
	{
		//la�o que percorre da �ltima posi��o
		//at� posi��o j do vetor
		for (int i = 4; i >= j; i--)
		{
			if (x[i] < x[i - 1])
			{
				aux = x[i];
				x[i] = x[i - 1];
				x[i - 1] = aux;
			}
		}
	}

}
